```python
def глас_господень(file):
    слова = file.read().split()
    откровение = sample(слов, randint(6, 150))  # Число зверя не употреби
    откровение[0] = откровение[0].upper()       # Да святится имя Твое
    return ' '.join(откровение) + '.'            # Аминь
```
